# Values used to calculate the hash in this folder name.
# Should not depend on the absolute path of the project itself.
#   - AGP: 8.13.0.
#   - $NDK is the path to NDK 29.0.14206865.
#   - $PROJECT is the path to the parent folder of the root Gradle build file.
#   - $ABI is the ABI to be built with. The specific value doesn't contribute to the value of the hash.
#   - $HASH is the hash value computed from this text.
NDK_PROJECT_PATH=null
APP_BUILD_SCRIPT=$PROJECT/terminal-emulator/src/main/jni/Android.mk
APP_ABI=$ABI
NDK_ALL_ABIS=$ABI
NDK_DEBUG=1
APP_PLATFORM=android-21
NDK_OUT=$PROJECT/terminal-emulator/build/intermediates/cxx/Debug/$HASH/obj
NDK_LIBS_OUT=$PROJECT/terminal-emulator/build/intermediates/cxx/Debug/$HASH/lib
APP_CFLAGS+=-std=c11
APP_CFLAGS+=-Wall
APP_CFLAGS+=-Wextra
APP_CFLAGS+=-Werror
APP_CFLAGS+=-Os
APP_CFLAGS+=-fno-stack-protector
APP_CFLAGS+=-Wl,--gc-sections
